\textbf{Листинг 2. Реализация конечного автомата (FSM) для теста цифровой гигиены}
\begin{lstlisting}[language=Python]
@bot.message_handler(func=lambda message: message.text == 'Тест цифровой гигиены')
def hygiene_start(message):
    user_states[message.chat.id] = STATE_HYGIENE_SCREEN
    markup = types.ReplyKeyboardMarkup(resize_keyboard=True, one_time_keyboard=True)
    markup.add('Менее 4 часов', 'От 4 до 8 часов', 'Более 8 часов')
    bot.send_message(message.chat.id, "Сколько часов в день вы проводите за экранами?", reply_markup=markup)

@bot.message_handler(func=lambda message: user_states.get(message.chat.id) == STATE_HYGIENE_SCREEN)
def hygiene_step1(message):
    score = 0
    if message.text == 'Менее 4 часов': score += 2
    elif message.text == 'От 4 до 8 часов': score += 1
    # Сохраняем промежуточный результат (упрощенно)
    user_states[f"{message.chat.id}_score"] = score
    
    user_states[message.chat.id] = STATE_HYGIENE_FILTER
    markup = types.ReplyKeyboardMarkup(resize_keyboard=True, one_time_keyboard=True)
    markup.add('Да, всегда', 'Иногда', 'Никогда')
    bot.send_message(message.chat.id, "Используете ли вы фильтры синего света перед сном?", reply_markup=markup)
\end{lstlisting}
